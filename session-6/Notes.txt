**************Format and mount an ebs volume************

---->lsblk
	////to check if the volume has any data
---->sudo file -s /dev/xvdf

---->sudo mkfs -t ext4 /dev/xvdf

---->sudo mkdir /newvolume

---->sudo mount /dev/xvdf /newvolume

---->cd /newvolume
---->df -h
